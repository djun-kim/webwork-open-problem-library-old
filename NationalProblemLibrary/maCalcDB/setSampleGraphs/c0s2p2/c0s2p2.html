sub pictIndex{    # this gives offsets into @pictID -- I was too laxy to do multidimensional arrays
	my ($i, @rest) =@_;
	my $j;
	for $j (0..$#rest) {
 		$rest[$j] = ($i-1)*8 +$rest[$j];
 	}
	@rest;
	}

$image = "${pictSet}.gif";

of the function.

@slice = &NchooseK(8,4);
